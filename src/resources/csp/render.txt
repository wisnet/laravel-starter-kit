    /**
     * Render an exception into a response.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  \Exception  $e
     * @return \Illuminate\Http\Response|\Symfony\Component\HttpFoundation\Response
     */
    public function render($request, Exception $e)
    {
        $this->container->singleton(\App\Policies\CSPPolicy::class, function ($app) {
            return new \App\Policies\CSPPolicy::class();
        });
        app(\App\Policies\CSPPolicy::class)->addDirective(\Spatie\Csp\Directive::SCRIPT, \Spatie\Csp\Keyword::UNSAFE_INLINE);
        app(\App\Policies\CSPPolicy::class)->addDirective(\Spatie\Csp\Directive::STYLE, \Spatie\Csp\Keyword::UNSAFE_INLINE);

        return parent::render($request, $exception);
    }
